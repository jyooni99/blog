## 긴 코드 블록 (스크롤 테스트)

```javascript
// 이것은 긴 코드 블록입니다. 스크롤이 필요할 수 있습니다.
function complexFunction(data) {
  const result = [];

  for (let i = 0; i < data.length; i++) {
    const item = data[i];

    if (item.type === "user") {
      result.push({
        id: item.id,
        name: item.name,
        email: item.email,
        role: item.role || "user",
        createdAt: item.createdAt,
        updatedAt: item.updatedAt,
      });
    } else if (item.type === "admin") {
      result.push({
        id: item.id,
        name: item.name,
        email: item.email,
        role: "admin",
        permissions: item.permissions || [],
        createdAt: item.createdAt,
        updatedAt: item.updatedAt,
      });
    }
  }

  return result.filter((item) => item.name && item.email);
}

// 더 많은 코드가 있습니다...
const users = [
  { id: 1, type: "user", name: "John", email: "john@example.com" },
  { id: 2, type: "admin", name: "Jane", email: "jane@example.com" },
];

const processedUsers = complexFunction(users);
console.log(processedUsers);
```
